CREATE PROCEDURE [dbo].[DeleteStudent]
(
    @StudentId int,
    @Message varchar(50) OUTPUT
)
AS 
BEGIN
    SET NOCOUNT ON;

    BEGIN TRY
        BEGIN TRANSACTION;

        BEGIN
            DELETE FROM Student
            WHERE StudentId = @StudentId;

            SET @Message = 'Deleted Successfully';
        END
        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        SET @Message = ERROR_MESSAGE();
        THROW;
    END CATCH;  
END
GO

CREATE PROCEDURE [dbo].[InsertStudent]
(
@FullName nvarchar(50),
@Address nvarchar(50),
@Email nvarchar(50),
@Contactno nvarchar(50),
@Gender nvarchar(50),
@EnrollmentDate datetime,
@Message varchar(50) OUTPUT,
@SID int OUTPUT
)
AS 
BEGIN

SET NOCOUNT ON;
BEGIN TRY
			BEGIN TRANSACTION;

BEGIN
if exists(select top 1 StudentId from Student where FullName=@FullName)
BEGIN
set @Message='Student Already Inserted'
set @SID=0
END
ELSE
BEGIN
INSERT INTO Student(FullName,[Address],Email,Contactno,Gender,EnrollmentDate )
values(@FullName,@Address,@Email,@Contactno,@Gender,@EnrollmentDate)

set @SID=SCOPE_IDENTITY();
set @Message='Saved Successfully' 
END
END
       
            COMMIT TRANSACTION;

        END TRY
        BEGIN CATCH
            ROLLBACK TRANSACTION;
set @Message=ERROR_MESSAGE();
THROW;
        END CATCH;  


   
END 
GO

GO
SET QUOTED_IDENTIFIER ON
GO
create Procedure [dbo].[PopulateSectionWiseProgramCount]
(
@GrantTypeId int,
@PhaseNumber int
)
as
Begin
if(@GrantTypeId=0)
Begin
select MS.SectionName, count(SPM.SubProgramId) as TotalProgram From SubProgramMaster SPM
right join MainSection MS on MS.MainSectionId=SPM.MainSectionId and SPM.Status=2 and SPM.PhaseStatus=@PhaseNumber
group by MS.SectionName
end
else
begin
select MS.SectionName, count(SPM.SubProgramId) as TotalProgram From SubProgramMaster SPM
right join MainSection MS on MS.MainSectionId=SPM.MainSectionId and SPM.Status=2 and SPM.PhaseStatus=@PhaseNumber
where SPM.GrantTypeId=@GrantTypeId
group by MS.SectionName

end
End
GO

CREATE PROCEDURE [dbo].[UpdateStudent]
(

    @StudentId int,
    @FullName nvarchar(50),
    @Address nvarchar(50),
    @Email nvarchar(50),
    @Contactno nvarchar(50),
    @Gender nvarchar(50),
    @EnrollmentDate datetime,
    @Message varchar(50) OUTPUT
)
AS 
BEGIN
    SET NOCOUNT ON;

    BEGIN TRY
        BEGIN TRANSACTION;

 
        BEGIN
            UPDATE Student
            SET FullName = @FullName,
                [Address] = @Address,
                Email = @Email,
                Contactno = @Contactno,
                Gender = @Gender,
                EnrollmentDate = @EnrollmentDate
            WHERE StudentId = @StudentId;

            SET @Message = 'Updated Successfully';
        END
        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        SET @Message = ERROR_MESSAGE();
        THROW;
    END CATCH;  
END
GO
